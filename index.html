const START = 1; const END = 240; const TOTAL = END - START + 1; const PATH = (i) => `frames/ezgif-frame-${String(i).padStart(3, '0')}.jpg`; let images = []; let currentFrame = 0; const canvas = document.getElementById("animation-canvas"); const ctx = canvas.getContext("2d"); // Resize function resize() { canvas.width = window.innerWidth; canvas.height = window.innerHeight; draw(currentFrame); } window.addEventListener("resize", resize); resize(); // Preload function preload() { return new Promise((resolve) => { let loaded = 0; for (let i = START; i <= END; i++) { const img = new Image(); img.src = PATH(i); img.onload = () => { if (++loaded === TOTAL) resolve(); }; img.onerror = () => { if (++loaded === TOTAL) resolve(); }; images.push(img); } }); } // Draw with zoom function draw(index) { const img = images[index]; if (!img) return; const zoom = 1 + index / TOTAL * 0.1; const width = canvas.width * zoom; const height = canvas.height * zoom; ctx.clearRect(0, 0, canvas.width, canvas.height); ctx.drawImage( img, (canvas.width - width) / 2, (canvas.height - height) / 2, width, height ); } // Scroll function onScroll() { const scroll = window.scrollY; const max = document.body.scrollHeight - window.innerHeight; let progress = scroll / max; // cinematic timing if (progress < 0.3) { progress *= 0.5; } else if (progress < 0.6) { progress = 0.15 + (progress - 0.3) * 1.5; } else { progress = 0.6 + (progress - 0.6) * 0.6; } const frame = Math.min(TOTAL - 1, Math.floor(progress * TOTAL)); if (frame !== currentFrame) { currentFrame = frame; requestAnimationFrame(() => draw(currentFrame)); } // text document.querySelectorAll(".overlay").forEach(el => { const start = parseFloat(el.dataset.start); const end = parseFloat(el.dataset.end); if (progress >= start && progress <= end) { el.classList.add("active"); } else { el.classList.remove("active"); } }); // navbar document.getElementById("navbar") .classList.toggle("scrolled", scroll > 50); } window.addEventListener("scroll", onScroll); // INIT + LOADER preload().then(() => { draw(0); const loader = document.getElementById("loader"); setTimeout(() => { loader.style.opacity = "0"; setTimeout(() => loader.style.display = "none", 1000); }, 500); });
